country_event = {
	id = chaos.221
	title = chaos.221.t
	desc = chaos.221.d

	is_triggered_only = yes

	option = {
		trigger = {
			321 = {
				OWNER = {
					check_variable = { var:num_owned_states > 1 } 
				}
			}
		}
		name = chaos.221.a
		321 = {
			OWNER = {
				set_temp_variable = { country_to_initiate = RAJ }
				custom_effect_tooltip = compliance_80_effect_tooltip

				#modified copy of vanilla instantiate_collaboration_government
				hidden_effect = {
					set_variable = { collaboration_formed@var:country_to_initiate = 1 }

					if = {
						# if country_to_initiate does not exist, simply release it as a puppet
						limit = {
							var:country_to_initiate = {
								exists = no
							}
						}
						var:country_to_initiate = { 
							set_variable = { collaboration_formed_by = PREV }
							transfer_state = 321
						}
						set_autonomy = {
							target = var:country_to_initiate
							autonomy_state = autonomy_collaboration_government
							freedom_level = 0.5
						}
					}
					else = {
						# if country_to_initiate exists, create a dynamic country as our new puppet

						create_dynamic_country = {
							original_tag = country_to_initiate

							set_variable = { collaboration_formed_by = PREV }
							set_temp_variable = { new_country = this }

							PREV = {
								var:new_country = {
									transfer_state = 321
								}

								puppet = new_country

								set_autonomy = {
									target = new_country
									autonomy_state = autonomy_collaboration_government
									freedom_level = 0.5
								}
							}
						}
					}
				}

				random_subject_country = {
					limit = {
						original_tag = RAJ
						is_dynamic_country = yes
					}
					add_manpower = 1
					every_owned_state = {
						limit = {
							NOT = { state = 321 }
						}
						transfer_state_to = PREV.PREV
					}
				}
			}
		}
	}
	option = {
		trigger = {
			321 = {
				OWNER = {
					check_variable = { var:num_owned_states < 2 } 
				}
			}
		}
		name = chaos.221.b
		add_political_power = -1
	}
}
